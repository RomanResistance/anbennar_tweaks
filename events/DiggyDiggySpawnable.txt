namespace = diggy_spawnable

#Began's Expedition
country_event = {
	id = diggy_spawnable.1
	title = diggy_spawnable.1.t
	desc = diggy_spawnable.1.d
	picture = ARTWORK_eventPicture
	goto = spawn_province
	
	fire_only_once = yes
	mean_time_to_happen = {
		years = 1
	}
	
	trigger = {
		is_year = 1460
		owns = 912
	}
	
	immediate = {
		hidden_effect = {
			random_province = {
				limit = {
					region = serpentreach_region
					is_empty = yes
					
					has_empty_adjacent_province = yes
					
				}
				
				save_event_target_as = spawn_province
			}
		}
	}
	
	#Let them go
	option = {
        name = diggy_spawnable.1.a
        ai_chance = { factor = 100 }
		event_target:spawn_province = {
			create_colony = 1000
		}
		
		event_target:spawn_province = {
			cede_province = H95
		}
		
		hidden_effect = {
			H95 = {
				country_event = {
					id = anb_miscevents.8
					days = 30
				}
				west_serpentspine_superregion = {
					discover_country = PREV
				}
				set_ruler = began
			}
		}
		
		adventurer_spawnable_starting_bonus = { tag = H95 }
	}
	#Play as them
	option = {
        name = diggy_spawnable.1.b
        ai_chance = { factor = 0 }
		event_target:spawn_province = {
			create_colony = 1000
		}
		
		event_target:spawn_province = {
			cede_province = H95
		}
		
		hidden_effect = {
			H95 = {
				country_event = { 
					id = anb_miscevents.8
					days = 30
				}
				set_ruler = began
			}
		}
		
		adventurer_spawnable_starting_bonus = { tag = H95 }
		switch_tag = H95
	}
}
