#Sornic Sons - Bloodgroves
country_event = {
	id = adventurerspawnables.1
	title = adventurerspawnables.1.t
	desc = adventurerspawnables.1.d
	picture = CARIBBEAN_PIRATE_FORT_eventPicture
	goto = spawn_province
	
	fire_only_once = yes
	
	trigger = {
		owns = 88 #Sornaire
		has_global_flag = new_world_discovered
		is_year = 1500
		bloodgroves_region = {
			is_empty = yes
			
			#has_discovered = ROOT
			
			has_empty_adjacent_province = yes
		}
	}
	
	mean_time_to_happen = {
		months = 300
		
		modifier = {
			factor = 0.1
			A09 = { exists = no }
		}
		
		modifier = {
			factor = 0.5
			any_country = {
				capital_scope = { continent = europe }
				bloodgroves_region = { has_discovered = ROOT }
			}
		}
		
		modifier = {
			factor = 0.5
			ai = no
		}
	}
	
	immediate = {
		hidden_effect = {
			random_province = {
				limit = {
					region = bloodgroves_region
					is_empty = yes
					
					has_empty_adjacent_province = yes
					
				}
				
				save_event_target_as = spawn_province
			}
			
			random_province = {
				limit = {
					region = bloodgroves_region
					is_empty = yes
					
					has_empty_adjacent_province = yes
					
					has_port = yes
				}
				
				save_event_target_as = spawn_province
			}		
		}
	}
	
	option = {
		name = adventurerspawnables.1.a
		
		event_target:spawn_province = {
			create_colony = 1000
		}
		
		event_target:spawn_province = {
			cede_province = H40
		}
		
		hidden_effect = {
			H40 = {
				country_event = {
					id = anb_miscevents.8
					days = 30
				}
			}
		}
		
		adventurer_spawnable_starting_bonus = { tag = H40 }
	}
	
	option = { # play as the pirates
		name = adventurerspawnables.1.b
		trigger = {
			ai = no
		}
		event_target:spawn_province = {
			create_colony = 1000
		}
		
		event_target:spawn_province = {
			cede_province = H40
		}
		
		hidden_effect = {
			H40 = {
				country_event = { 
					id = anb_miscevents.8
					days = 30
				}
			}
		}
		
		adventurer_spawnable_starting_bonus = { tag = H40 }
		switch_tag = H40
	}
	

}
