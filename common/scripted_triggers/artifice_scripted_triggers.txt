over_artifice_capacity = {
	OR = {
		has_country_modifier = artificers_over_artifice_capacity
		NOT = {
			check_variable = {
				which = MaxArtificePoints
				value = 0
			}
		}
	}
}

artifice_has_points_for_basic = {
	custom_trigger_tooltip = {
		tooltip = cost_5_artifice_capacity_to_maintain_tt
		check_variable = {
			which = MaxArtificePoints
			value = 5
		}
	}
}
